Código (Pipeline)

import pandas as pd

df = pd.read_csv("orders.csv")
df["order_date"] = pd.to_datetime(df["order_date"])
df = df.dropna(subset=["order_id", "customer_id", "total"])
df = df[df["total"] > 0]

df["month"] = df["order_date"].dt.to_period("M").dt.to_timestamp()

revenue_by_month = (
    df.groupby("month", as_index=False)["total"]
      .sum()
      .rename(columns={"total": "revenue"})
)

ticket_medio = (
    df.groupby("month", as_index=False)["total"]
      .mean()
      .rename(columns={"total": "avg_ticket"})
)

WITH base AS (
  SELECT
    date_trunc('month', order_date) AS month,
    category,
    total
  FROM orders
  WHERE total > 0
)
SELECT
  month,
  SUM(total) AS revenue,
  AVG(total) AS avg_ticket,
  SUM(total) / NULLIF(COUNT(*), 0) AS avg_ticket_alt
FROM base
GROUP BY month
ORDER BY month;
